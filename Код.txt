using Avalonia.Controls;
using System;
using Avalonia.Interactivity;
using Npgsql;
using System.Threading.Tasks;
namespace ShooesApp.Views;

public partial class MainWindow : Window
{

    public MainWindow()
    {
        InitializeComponent();
     
    }
    private async void LoginButton_Click(object? sender, RoutedEventArgs e)

    {

        var login = UserBox.Text;

        var password = PassBox.Text;

        if (string.IsNullOrEmpty(login) || string.IsNullOrEmpty(password))

        {

            await MessageBox("Введите логин и пароль");

            return;

        }



        var connStr = "Host=localhost;Port=5432;Username=postgres;Password=1;Database=botinki";

        await using var conn = new NpgsqlConnection(connStr);

        await conn.OpenAsync();



        var query = @"
SELECT user_id, role_id, login, password, first_name, last_name, middle_name
FROM users 
WHERE login = @l AND password = @p";



        var cmd = new NpgsqlCommand(query, conn);

        cmd.Parameters.AddWithValue("l", login);

        cmd.Parameters.AddWithValue("p", password);



        using var reader = await cmd.ExecuteReaderAsync();



        if (await reader.ReadAsync())

        {

            var roleIdObj = reader["role_id"];

            var roleId = Convert.ToInt32(roleIdObj);



            if (roleId == 1)

            {

                // Менеджер 

                new MenuWindow().Show();

                this.Close();

            }

            else if (roleId == 2)

            {

                // Менеджер (по данным из базы) 

                new MenuWindow().Show();

                this.Close();

            }

            else if (roleId == 3)

            {

                // Авторизированный клиент 

                var firstName = reader.IsDBNull(reader.GetOrdinal("first_name")) ? string.Empty : reader.GetString(reader.GetOrdinal("first_name"));
                var lastName = reader.IsDBNull(reader.GetOrdinal("last_name")) ? string.Empty : reader.GetString(reader.GetOrdinal("last_name"));
                var middleName = reader.IsDBNull(reader.GetOrdinal("middle_name")) ? string.Empty : reader.GetString(reader.GetOrdinal("middle_name"));
                var fullName = $"{lastName} {firstName} {middleName}".Trim();
                new ClientWindow(fullName).Show();

                this.Close();

            }

            else

            {

                await MessageBox("Неизвестная роль пользователя");

            }

        }

        else

        {

            await MessageBox("Неверный логин или пароль");

        }

    }

    private async Task MessageBox(string text)

    {

        var win = new Window { Content = new TextBlock { Text = text }, Width = 300, Height = 100 };

        await win.ShowDialog(this);

    }

    private void ExitButton_Click(object? s, RoutedEventArgs e)
    {
        Close();
    }
    private void GuestButton_Click(object? s, RoutedEventArgs e)
    {
        var guestwin = new GuestWindow();
        guestwin.Show();
        Close();

    }


}

public partial class GuestWindow : Window

    private readonly List<Product> _products;

    public GuestWindow()
    {
        InitializeComponent();
        _products = LoadProducts();
        foreach (var p in _products)
        {
            CardsPanel.Children.Add(CreateCard(p));
        }
    }
    private void BackButton_Click(object? sender, RoutedEventArgs e)
    {
        new MainWindow().Show();
        Close();
    }

    private static Control CreateCard(Product product)
    {
        var photoBox = new Border
        {
            BorderThickness = new Thickness(1),
            BorderBrush = Brushes.Black,
            Margin = new Thickness(0, 0, 8, 0),
            Width = 120,
            Height = 150,
            Child = new Image
            {
                Source = GetPhoto(product.Photo),
                Stretch = Stretch.Uniform,
                VerticalAlignment = VerticalAlignment.Center,
                HorizontalAlignment = HorizontalAlignment.Center
            }
        };
        Grid.SetColumn(photoBox, 0);

        var infoSection = new StackPanel
        {
            Spacing = 6,
            Children =
            {
                new TextBlock { Text = $"{product.Category} | {product.ProductName}", FontSize = 18, FontWeight = FontWeight.SemiBold },
                new TextBlock { Text = $"Описание: {product.Description}" },
                new TextBlock { Text = $"Производитель: {product.Manufacturer}" },
                new TextBlock { Text = $"Поставщик: {product.Supplier}" },
                CreatePricePanel(product),
                new TextBlock { Text = $"Единица измерения: {product.Unit}" },
                new TextBlock { Text = $"Количество на складе: {product.StockQty}" }
            }
        };
        Grid.SetColumn(infoSection, 1);

        var discountBox = new Border
        {
            BorderThickness = new Thickness(1),
            BorderBrush = Brushes.Black,
            Padding = new Thickness(8),
            Child = new StackPanel
            {
                Spacing = 4,
                VerticalAlignment = VerticalAlignment.Center,
                Children =
                {
                    new TextBlock { Text = "Действующая скидка", FontWeight = FontWeight.SemiBold, TextAlignment = TextAlignment.Center },
                    new TextBlock { Text = $"{product.Discount}%", FontSize = 24, FontWeight = FontWeight.Bold, TextAlignment = TextAlignment.Center }
                }
            }
        };
        Grid.SetColumn(discountBox, 2);

        var grid = new Grid
        {
            ColumnDefinitions = new ColumnDefinitions("120,*,120"),
            Children = { photoBox, infoSection, discountBox }
        };

        return new Border
        {
            Padding = new Thickness(8),
            CornerRadius = new CornerRadius(6),
            BorderThickness = new Thickness(1),
            BorderBrush = Brushes.Black,
            Background = product.StockQty == 0
                ? Brushes.LightBlue
                : product.Discount > 15
                    ? new SolidColorBrush(Color.Parse("#2E8B57"))
                    : Brushes.White,
            Child = grid
        };
    }

    private static IImage GetPhoto(string? photo)
    {
        var defaultPath = @"C:\Users\brawl\OneDrive\Рабочий стол\demo\Botinki\Assets\Icon.png";
        var filename = string.IsNullOrWhiteSpace(photo) ? "Icon.png" : photo;
        var candidates = new[]
        {
            Path.Combine(AppContext.BaseDirectory, "Assets", filename),
            Path.Combine(Directory.GetCurrentDirectory(), "Assets", filename),
            Path.Combine(AppContext.BaseDirectory, filename),
            defaultPath
        };

        foreach (var candidate in candidates)
        {
            if (!string.IsNullOrWhiteSpace(candidate) && File.Exists(candidate))
            {
                return new Bitmap(candidate);
            }
        }

        return new Bitmap(defaultPath);
    }

    private static Control CreatePricePanel(Product product)
    {
        var pricePanel = new StackPanel { Orientation = Orientation.Horizontal, Spacing = 8 };
        if (product.Discount > 0)
        {
            pricePanel.Children.Add(new TextBlock { Text = $"{product.Price:F2} ₽", Foreground = Brushes.Red, TextDecorations = TextDecorations.Strikethrough });
            pricePanel.Children.Add(new TextBlock { Text = $"{product.FinalPrice:F2} ₽", Foreground = Brushes.Black });
        }
        else
        {
            pricePanel.Children.Add(new TextBlock { Text = $"{product.Price:F2} ₽", Foreground = Brushes.Black });
        }
        return pricePanel;
    }

    private static List<Product> LoadProducts()
    {
        var products = new List<Product>();
        var connStr = "Host=localhost;Port=5432;Username=postgres;Password=1;Database=botinki";
        using var conn = new NpgsqlConnection(connStr);
        conn.Open();
        using var cmd = new NpgsqlCommand("select article,product_name,unit,price,supplier,manufacturer,category,discount,stock_qty,description,photo from product_names order by article", conn);
        using var reader = cmd.ExecuteReader();
        while (reader.Read())
        {
            products.Add(new Product(
                reader.GetString(0),
                reader.GetString(1),
                reader.IsDBNull(2) ? string.Empty : reader.GetString(2),
                reader.GetDecimal(3),
                reader.IsDBNull(4) ? string.Empty : reader.GetString(4),
                reader.IsDBNull(5) ? string.Empty : reader.GetString(5),
                reader.IsDBNull(6) ? string.Empty : reader.GetString(6),
                reader.IsDBNull(7) ? 0 : reader.GetInt32(7),
                reader.IsDBNull(8) ? 0 : reader.GetInt32(8),
                reader.IsDBNull(9) ? string.Empty : reader.GetString(9),
                reader.IsDBNull(10) ? null : reader.GetString(10)
            ));
        }
        return products;
    }

    private record Product(string Article, string ProductName, string Unit, decimal Price, string Supplier, string Manufacturer, string Category, int Discount, int StockQty, string Description, string? Photo)
    {
        public decimal FinalPrice => Price * (100 - Discount) / 100;
    }
}
